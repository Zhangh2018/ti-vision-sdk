845 static inline void CPDMA_appendPacketQueue(CPDMA_PacketQueue *pCPDMAPacketQueueDest, const CPDMA_PacketQueue *pCPDMAPacketQueueSrc)
846 {
847  if (pCPDMAPacketQueueDest->pHead == NULL)
848  {
849  /* Destination packet queue is empty, initialize it with the source packet queue */
850  pCPDMAPacketQueueDest->pHead = pCPDMAPacketQueueSrc->pHead;
851  pCPDMAPacketQueueDest->count = pCPDMAPacketQueueSrc->count;
852  }
853  else
854  {
855  /* Destination queue is not empty, add source queue contents onto end of the destination queue */
856  pCPDMAPacketQueueDest->pTail->pNextPacket = pCPDMAPacketQueueSrc->pHead;
857  pCPDMAPacketQueueDest->count += pCPDMAPacketQueueSrc->count;
858  }
859  pCPDMAPacketQueueDest->pTail = pCPDMAPacketQueueSrc->pTail;
860 }
