    36    /*!
    37     *  ======== TransportRpmsg ========
    38     *  Transport for MessageQ that uses vring structures.
    39     *
    40     *  This is a {@link ti.sdo.ipc.MessageQ} transport that utilizes
    41     *  a pair of vrings (see Linux virtio) to communicate with a remote processor.
    42     *
    43     */
    44    
    45    @InstanceFinalize
    46    @InstanceInitError
    47    
    48    
    49    module TransportRpmsg inherits ti.sdo.ipc.interfaces.IMessageQTransport
    50    {
    51    
    52    instance:
    53    
    54        /*!
    55         *  ======== sharedAddr ========
    56         *  Address in shared memory where this instance will be placed
    57         *
    58         */
    59        config Ptr sharedAddr = null;
    60    
    61    
    62    internal:
    63    
    64        /*! Instance state structure */
    65        struct Instance_State {
    66            UInt16       priority;           /* priority to register             */
    67            UInt16       remoteProcId;       /* dst proc id                      */
    68            Ptr          msgqHandle;         /* RPMessage Handle              */
    69        }
    70    }
