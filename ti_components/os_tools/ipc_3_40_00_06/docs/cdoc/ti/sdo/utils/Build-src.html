    33    /*
    34     *  ======== Build.xdc ========
    35     *  @_nodoc
    36     *  metaonly module to support building various package/product libraries
    37     *
    38     */
    39    
    40    /*!
    41     *  ======== Build ========
    42     */
    43    
    44    @Template("./Build.xdt")
    45    metaonly module Build
    46    {
    47    
    48        /*!
    49         *  ======== doBuild ========
    50         */
    51        config Bool doBuild = true;
    52    
    53        /*!
    54         *  ======== libDir ========
    55         */
    56        config String libDir = null;
    57    
    58        /*!
    59         *  ======== getDefaultCustomCCOpts ========
    60         */
    61        String getDefaultCustomCCOpts();
    62    
    63        /*!
    64         *  ======== getDefs ========
    65         *  Get the compiler -D options necessary to build
    66         */
    67        String getDefs();
    68    
    69        /*!
    70         *  ======== getCFiles ========
    71         *  Get the library C source files.
    72         */
    73        String getCFiles(String target);
    74    
    75        /*!
    76         *  ======== getAsmFiles ========
    77         *  Get the library Asm source files.
    78         */
    79        Any getAsmFiles(String target);
    80    
    81        /*
    82         *  ======== buildLibs ========
    83         *  This function generates the makefile goals for the libraries
    84         *  produced by a ti.sysbios package.
    85         */
    86        function buildLibs(objList, relList, filter, xdcArgs);
    87    
    88        /*!
    89         *  ======== getLibs ========
    90         *  Common getLibs() for all ipc packages.
    91         */
    92        function getLibs(pkg);
    93    }
