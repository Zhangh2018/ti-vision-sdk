    37    /*!
    38     *  ======== Http ======== 
    39     *  NDK HTTP server settings and configuration.
    40     *  
    41     *  The Http module is used to create an HTTP server on the target processor.
    42     *  Using the Http module, the user is able easily set up and add an HTTP
    43     *  server to their application.
    44     *  
    45     *  Summary of Steps:
    46     *  
    47     *  The following is a summary of required steps to add a HTTP server to an NDK
    48     *  application:
    49     *  
    50     *  1. Use the Http module
    51     *  
    52     *  2. Create a Http parameter structure and configure its properties.
    53     *  
    54     *  3. Create an Http module instance based on the properties set in the
    55     *     configuration structure.
    56     *  
    57     *  Detailed Steps:
    58     *  
    59     *  1. To add an HTTP server using the Http module, it's first necessary to use
    60     *     the Http module in the application RTSC configuration file:
    61     *  @p(code)
    62     *  var Http = xdc.useModule('ti.ndk.config.Http');
    63     *  @p
    64     *  2. After the Http module is used, it can then be referenced to create a 
    65     *     Http parameter structure, through which the properties of the HTTP
    66     *     server being added can be configured:
    67     *  @p(code)
    68     *  var httpParams = new Http.Params();
    69     *  httpParams.port = 1003;
    70     *  httpParams.maxCon = 11;
    71     *  @p
    72     *  3. Once the properties of the HTTP server have been set (via the parameter
    73     *     structure), an Http instance may be created based on those parameters:
    74     *  @p(code)
    75     *  var httpInst = Http.create(httpParams);
    76     *  @p
    77     *  
    78     *  As the above steps indicate, the Http module is configured on a per
    79     *  instance basis.  The steps above detail how to configure a single Http
    80     *  instance, but, it is also possible to create several Http servers, all
    81     *  running on the same target processor, which accept connections on
    82     *  different ports.
    83     *  
    84     *  For example, the above steps illustrate how to add a HTTP server which
    85     *  accepts connections on port 1003.  To add a second HTTP server, which
    86     *  accepts connections on the port 1004, the following code could be
    87     *  added:
    88     *  @p(code)
    89     *  var http2Params = new Http.Params();
    90     *  http2Params.port = 1004;
    91     *  var http2Inst = Http.create(http2Params);
    92     *  @p
    93     */  
    94    
    95    @Template("./Http.xdt")
    96    
    97    metaonly module Http {
    98    
    99        /*! Function signature for CISARGS struct service reporting function */
   100        typedef Void (*httpServiceFxn)(Int, Int, Int, Void *);
   101    
   102        /*! Type used to specify bits in mode */
   103        typedef Bits16 CisFlags;
   104    
   105        /*! Specifies if the IfIdx field is valid. */
   106        const CisFlags CIS_FLG_IFIDXVALID        = 0x0001;
   107    
   108        /*!
   109         *  Requests that IfIdx be resolved to an IP address before service
   110         *  execution is initiated.
   111         */
   112        const CisFlags CIS_FLG_RESOLVEIP         = 0x0002;
   113    
   114        /*! Specifies that the service should be invoked by IP address */
   115        const CisFlags CIS_FLG_CALLBYIP          = 0x0004;
   116    
   117        /*!
   118         *  A service that is dependent on a valid IP address (as determined by the
   119         *  RESOLVEIP flag) is shut down if the IP address becomes invalid.
   120         *
   121         *  When this flag is set, the service will be restarted when a new address
   122         *  becomes available. Otherwise; the service will not be restarted.
   123         */
   124        const CisFlags CIS_FLG_RESTARTIPTERM     = 0x0008;
   125    
   126        /*!
   127         *  ======== create ========
   128         *  Creates an Http instance.
   129         */
   130        create();
   131    
   132    instance:
   133    
   134        /*!
   135         *  Specifies the maximum number of hosts that may connect to the HTTP
   136         *  server running on this port (1 - 24).
   137         */
   138        config Int maxCon = 8;
   139    
   140        /*!
   141         *  Specifies the port number which this HTTP server will accept
   142         *  connections.
   143         */ 
   144        config Int port = 80;
   145    
   146        /*!
   147         *  Set of flags which represent the desired behavior of the HTTP Server.
   148         *  
   149         *  The following flag values may be set either individually, or by or-ing
   150         *  flags together:
   151         * @p(blist)
   152         *    - CIS_FLG_IFIDXVALID - specifies if the IfIdx field is valid.
   153         *  
   154         *    - CIS_FLG_RESOLVEIP - Requests that IfIdx be resolved to an IP
   155         *      address before service execution is initiated.
   156         *  
   157         *    - CIS_FLG_CALLBYIP - Specifies that the service should be invoked by
   158         *      IP address
   159         *  
   160         *    - CIS_FLG_RESTARTIPTERM - A service that is dependent on a valid IP
   161         *      address.
   162         * @p
   163         */
   164        config CisFlags mode = CIS_FLG_IFIDXVALID;
   165    
   166        /*!
   167         *  The physical device index on which the HTTP server shall be
   168         *  executed.  Must be greater than zero.
   169         */
   170        config Int ifIdx = 1;
   171    
   172        /*! 
   173         *  The IP address on which to initiate this service.
   174         *  
   175         *  To accept a connection from any IP, specify INADDR_ANY.
   176         */ 
   177        config String ipAddr = "INADDR_ANY";
   178    
   179        /*! HTTP service reporting function. */
   180        config httpServiceFxn pCbSrv = "&ti_ndk_config_Global_serviceReport";
   181    
   182        /*! CGI functions and names */
   183        config string cgiFunctions[string];
   184    
   185        /*! @_nodoc
   186         *
   187         * internal use only.  Intermediate variables mapped to Grace checkboxes,
   188         * used to update 'mode' flags variable.
   189         */
   190        config Bool IfIdXValid = false;
   191    
   192        /*! @_nodoc
   193         *
   194         * internal use only.  Intermediate variables mapped to Grace checkboxes,
   195         * used to update 'mode' flags variable.
   196         */
   197        config Bool ResolveIP = false;
   198    
   199        /*! @_nodoc
   200         *
   201         * internal use only.  Intermediate variables mapped to Grace checkboxes,
   202         * used to update 'mode' flags variable.
   203         */
   204        config Bool CallByIP = false;
   205    
   206        /*! @_nodoc
   207         *
   208         * internal use only.  Intermediate variables mapped to Grace checkboxes,
   209         * used to update 'mode' flags variable.
   210         */
   211        config Bool RestartIPTerm = false;
   212    }
